WINDOWS L
OLED_CLEAR
OLED_UPDATE
SWC_FILL 255 0 0

VAR $KEY_STATE = 2

WHILE TRUE
    $KEY_STATE = $_READKEY
    IF $KEY_STATE == 3 THEN
	OLED_CLEAR
	OLED_CURSOR 10 10
	OLED_PRINT UNLOCKED
	OLED_UPDATE
	SWC_FILL 0 255 0
	DELAY 2000
	SWC_RESET 99
	LBREAK
    ELSE 
	WINDOWS L
	END_IF
END_WHILE
